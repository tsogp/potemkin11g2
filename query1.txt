UPDATE user SET created_at = substr(created_at, 1, 4) || '-' || substr(created_at, 9, 2) || '-' || substr(created_at, 6, 2)

-- последний зарегистрированный пользователь
SELECT username, MAX(created_at) AS 'last_register' FROM user

-- выборка по году рождения
SELECT id, username, substr(birth_date, 1, 4) as 'birth_year' FROM user GROUP BY id

-- выборка пользовательлей, зарегистрированных не позднее чем за 2 последних месяца
